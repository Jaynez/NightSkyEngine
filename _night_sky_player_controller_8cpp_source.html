39// Called every frame
40void ANightSkyPlayerController::Tick(float DeltaTime)
41{
42 Super::Tick(DeltaTime);
43 if (NetworkPawn != nullptr)
44 {
45 const int PlayerIndex = Cast<UNightSkyGameInstance>(GetGameInstance())->PlayerIndex;
46 TArray<ANetworkPawn*> NetworkPawns;
47 for (TActorIterator<ANetworkPawn> It(GetWorld()); It; ++It)
48 {
49 NetworkPawns.Add(*It);
50 }
51 if (NetworkPawns.Num() > 1)
52 {
53 if (PlayerIndex == 0)
54 {
55 SendGgpo(NetworkPawns[1], true);
56 }
57 else
58 {
59 SendGgpo(NetworkPawns[0], false);
60 }
61 }
62 if (NetworkPawn->bRematchAccepted && bRematch)
63 {
64 NetworkPawn->bRematchAccepted = false;
65 bRematch = false;
66 Cast<ANightSkyGameState>(GetWorld()->GetGameState())->MatchInit();
67 }
68 }
69}

276void ANightSkyPlayerController::SendGgpo(ANetworkPawn* InNetworkPawn, bool Client) const
277{
278 if(InNetworkPawn->FighterMultiplayerRunner==nullptr)//TODO: CHECK IF MULTIPLAYERRUNNER IS SPAWNED BEFORE THIS, IF SO DO THIS IN BEGINPLAY
279 {
280 TArray<AActor*> FoundFighterGameStates;
281 UGameplayStatics::GetAllActorsOfClass(GetWorld(), AFighterMultiplayerRunner::StaticClass(), FoundFighterGameStates);
282 if(FoundFighterGameStates.Num()>0)
283 {
284 InNetworkPawn->FighterMultiplayerRunner = Cast<AFighterMultiplayerRunner>(FoundFighterGameStates[0]);

293 if(Client)
294 {
295 InNetworkPawn->SendGgpoToClient(SendVal->GetValue());
296 }
297 else
298 {
299 InNetworkPawn->SendGgpoToServer(SendVal->GetValue());
300 }
301 InNetworkPawn->FighterMultiplayerRunner->connectionManager->sendSchedule.Empty();
302 //InNetworkPawn->FighterMultiplayerRunner->connectionManager->sendSchedule.RemoveNode(SendVal);
303 }
304 }

307void ANightSkyPlayerController::SendBattleData()
308{
309 int PlayerIndex = 0;
310 if (GetWorld()->GetNetMode() == NM_Client)
311 PlayerIndex = Cast<UNightSkyGameInstance>(GetGameInstance())->PlayerIndex = 1;
312 TArray<ANetworkPawn*> NetworkPawns;
313 for (TActorIterator<ANetworkPawn> It(GetWorld()); It; ++It)
314 {
315 NetworkPawns.Add(*It);
316 }
317 if (NetworkPawns.Num() > 1)
318 {
319 const UNightSkyGameInstance* GameInstance = Cast<UNightSkyGameInstance>(GetGameInstance());
320 if (PlayerIndex == 0)
321 {
322 NetworkPawns[1]->ClientGetBattleData(GameInstance->BattleData);
323 }
324 else
325 {
326 NetworkPawns[0]->ServerGetBattleData(GameInstance->BattleData);
327 }
