77int32 ANightSkyAIController::CheckAttackWeight(const UState* State) const
78{
79 const int32 AttackXBeginRange = State->CPUData.AttackXBeginRange;
80 const int32 AttackXEndRange = State->CPUData.AttackXEndRange;
81 const int32 AttackYBeginRange = State->CPUData.AttackYBeginRange;
82 const int32 AttackYEndRange = State->CPUData.AttackYEndRange;
83
84 if (!Player->CheckEnemyInRange(AttackXBeginRange, AttackXEndRange, AttackYBeginRange, AttackYEndRange)) return 0;
85
86 int32 Weight = 75;
87
88 if (State->CPUData.bPunish)
89 {
90 bool bInPunishRange = false;
91 switch (State->CPUData.PunishRange)
92 {
93 case RAN_Near:
94 bInPunishRange = GetEnemyDistanceX() <= 360000;
95 break;
96 case RAN_Mid:
97 bInPunishRange = GetEnemyDistanceX() > 360000 && GetEnemyDistanceX() <= 1080000;
98 break;
99 case RAN_Far:
100 bInPunishRange = GetEnemyDistanceX() > 720000;
101 break;
102 default:
103 break;
104 }
105
106 Weight += bInPunishRange ? 25 : 0;
107 }
108
109 if (Player->Enemy->StoredStateMachine.CurrentState->StateType == EStateType::Hitstun)
110 {
111 if (State->CPUData.bThrow) Weight = 0;
112 if (State->CPUData.bCombo) Weight += 25;
113 }
114 else if (Player->IsEnemyBlocking())
115 {
116 if (State->CPUData.bBlockstring) Weight += 25;
117 if (Player->Enemy->GetCurrentStateName() == State_Universal_StandBlock)
118 {
119 if (State->CPUData.BlockType == BLK_Low) Weight += 25;
120 }
121 else if (Player->Enemy->GetCurrentStateName() == State_Universal_CrouchBlock)
122 {
123 if (State->CPUData.BlockType == BLK_High) Weight += 25;
124 }
125 else if (State->CPUData.BlockType == BLK_None) Weight += 25;
126
127 if (State->CPUData.bThrow)
128 {
129 if (Player->Enemy->CheckIsStunned()) Weight = 0;
130 else Weight += 25;
131 }
132 }
133 else
134 {
135 if (Player->Enemy->CheckIsAttacking())
136 {
137 switch (State->CPUData.AttackSpeed)
138 {
139 case ASPD_Fast:
140 Weight += 25;
141 break;
142 case ASPD_Medium:
143 break;
144 case ASPD_Slow:
145 Weight -= 25;
146 break;
147 default: break;
148 }
149 if (State->CPUData.bInvuln) Weight += 25;
150 }
151 else if (Player->Enemy->IsEnemyAttackState() && State->CPUData.bPunish) Weight += 25;
152 }
153
154 if (Player->PosY < Player->Enemy->PosY && State->CPUData.bAntiAir) Weight += 25;
155
156 if (State->CPUData.bUsesResource) Weight -= 25;
157 if (State->CPUData.bBigDamage) Weight += 25;
158
159 return Weight;
160}

172void ANightSkyAIController::Tick(float DeltaTime)
173{
174 Super::Tick(DeltaTime);
175
176 if (!Player) Player = GetPawn<APlayerObject>();
177 if (GameState->BattleState.TimeUntilRoundStart) return;
178 if (Player->PlayerFlags & PLF_RoundWinInputLock) return;
179
180 if (Player->StoredStateMachine.CurrentState->StateType == EStateType::ForwardWalk || Player->StoredStateMachine.
181 CurrentState->StateType == EStateType::ForwardDash)
182 {
183 Player->Inputs = INP_Right;
184 if (Player->Direction == DIR_Left) Player->Inputs = INP_Left;
185 }
186 else if (Player->StoredStateMachine.CurrentState->StateType == EStateType::BackwardWalk)
187 {
188 Player->Inputs = INP_Left;
189 if (Player->Direction == DIR_Left) Player->Inputs = INP_Right;
190 }
191
192 TArray<UState*> AvailableStates = {};
193
194 for (auto State : Player->StoredStateMachine.States)
195 {
196 if (Player->CanEnterState(State)) AvailableStates.Add(State);
197 }
198
199 TMap<UState*, int32> StateWeights;
200 for (auto State : AvailableStates)
201 {
202 int32 Weight = 0;
203 switch (State->StateType)
204 {
205 case EStateType::Standing:
206 break;
207 case EStateType::Crouching:
208 break;
209 case EStateType::NeutralJump:
210 break;
211 case EStateType::ForwardJump:
212 break;
213 case EStateType::BackwardJump:
214 break;
215 case EStateType::ForwardWalk:
216 Weight = CheckFWalkWeight();
217 break;
218 case EStateType::BackwardWalk:
219 Weight = CheckBWalkWeight();
220 break;
221 case EStateType::ForwardDash:
222 Weight = CheckFDashWeight();
223 break;
224 case EStateType::BackwardDash:
225 break;
226 case EStateType::ForwardAirDash:
227 break;
228 case EStateType::BackwardAirDash:
229 break;
230 case EStateType::NormalAttack:
231 case EStateType::SpecialAttack:
232 case EStateType::SuperAttack:
233 Weight = CheckAttackWeight(State);
234 break;
235 case EStateType::Hitstun:
236 break;
237 case EStateType::Blockstun:
238 break;
239 case EStateType::Tech:
240 Weight = 250;
241 break;
242 case EStateType::Burst:
243 Weight = CheckBurstWeight();
244 break;
245 case EStateType::Tag:
246 break;
247 case EStateType::Custom:
248 break;
249 }
250
251 if (Weight > 250) Weight = 250;
252 if (Weight < 0) Weight = 0;
253
254 if (Weight == 0) continue;
255 StateWeights.Add(State, Weight);
256 }
257
258 StateWeights.ValueSort([](const int32 A, const int32 B)
259 {
260 return A > B;
261 });
262
263 TArray<UState*> Keys;
264 StateWeights.GetKeys(Keys);
265
266 if (Keys.Num())
267 {
268 Player->SetStateForCPU(Keys[0]->Name);
269 }
270}
