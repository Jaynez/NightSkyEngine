105void UNightSkyGameInstance::FindReplays()
106{
107 ReplayList.Empty();
108 for (int i = 0; i < MaxReplays; i++)
109 {
110 FString ReplayName = "REPLAY";
111 ReplayName.AppendInt(i);
112 if (!UGameplayStatics::DoesSaveGameExist(ReplayName, 0))
113 {
114 continue;
115 }
116 ReplayList.Add(Cast<UReplaySaveInfo>(UGameplayStatics::LoadGameFromSlot(ReplayName, 0)));
117 ReplayList.Last()->ReplayIndex = i;
118 if (ReplayList.Last()->Version != BattleVersion)
119 {
120 ReplayList.Pop();
121 UGameplayStatics::DeleteGameInSlot(ReplayName, 0);
122 }
123 }
124 BP_OnFindReplaysComplete(ReplayList);
