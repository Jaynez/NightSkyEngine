38void UNSESessionSubsystem::CreateSession(int32 NumPublicConnections, bool bIsLANMatch)
39{
40 const IOnlineSessionPtr SessionInterface = Online::GetSessionInterface(GetWorld());
41 if (!SessionInterface.IsValid())
42 {
43 OnCreateSessionCompleteDelegate.Broadcast(false);
44 return;
45 }
46
47 LastSessionSettings = MakeShareable(new FOnlineSessionSettings());
48 LastSessionSettings->NumPrivateConnections = 0;
49 LastSessionSettings->NumPublicConnections = NumPublicConnections;
50 LastSessionSettings->bAllowInvites = true;
51 LastSessionSettings->bAllowJoinInProgress = true;
52 LastSessionSettings->bAllowJoinViaPresence = true;
53 LastSessionSettings->bAllowJoinViaPresenceFriendsOnly = false;
54 LastSessionSettings->bIsDedicated = false;
55 LastSessionSettings->bUsesPresence = true;
56 LastSessionSettings->bIsLANMatch = bIsLANMatch;
57 LastSessionSettings->bShouldAdvertise = true;
58
59 auto GameVersion = Cast<UNightSkyGameInstance>(GetGameInstance())->GameVersion;
60
61 LastSessionSettings->Set(SETTING_SESSIONKEY, GameVersion, EOnlineDataAdvertisementType::ViaOnlineServiceAndPing);
62
63 CreateSessionCompleteDelegateHandle = SessionInterface->AddOnCreateSessionCompleteDelegate_Handle(
64 CreateSessionCompleteDelegate);
65
66 const ULocalPlayer* localPlayer = GetWorld()->GetFirstLocalPlayerFromController();
67 if (!SessionInterface->CreateSession(*localPlayer->GetPreferredUniqueNetId(), NAME_GameSession,
68 *LastSessionSettings))
69 {
70 SessionInterface->ClearOnCreateSessionCompleteDelegate_Handle(CreateSessionCompleteDelegateHandle);
71
72 OnCreateSessionCompleteDelegate.Broadcast(false);
73 }
74}

238void UNSESessionSubsystem::OnFindSessionsComplete(bool bSuccess)
239{
240 if (const IOnlineSessionPtr SessionInterface = Online::GetSessionInterface(GetWorld()))
241 {
242 SessionInterface->ClearOnFindSessionsCompleteDelegate_Handle(FindSessionsCompleteDelegateHandle);
243 }
244
245 if (LastSessionSearch->SearchResults.Num() <= 0)
246 {
247 OnFindSessionsCompleteEvent.Broadcast(TArray<FNSESessionInfo>(), bSuccess);
248 return;
249 }
250
251 TArray<FNSESessionInfo> Sessions{};
252 for (auto SearchResult : LastSessionSearch->SearchResults)
253 {
254 Sessions.Add(FNSESessionInfo(
255 SearchResult, SearchResult.Session.OwningUserName, SearchResult.Session.NumOpenPrivateConnections,
256 SearchResult.Session.NumOpenPublicConnections
257 ));
258 }
259
260 OnFindSessionsCompleteEvent.Broadcast(Sessions, bSuccess);
261}

275bool UNSESessionSubsystem::TryTravelToCurrentSession()
276{
277 const IOnlineSessionPtr SessionInterface = Online::GetSessionInterface(GetWorld());
278 if (!SessionInterface.IsValid())
279 {
280 return false;
281 }
282
283 FString ConnectString;
284 if (!SessionInterface->GetResolvedConnectString(NAME_GameSession, ConnectString))
285 {
286 return false;
287 }
288
289 APlayerController* PlayerController = GetWorld()->GetFirstPlayerController();
290 PlayerController->ClientTravel(ConnectString, TRAVEL_Absolute);
291 return true;
292}
